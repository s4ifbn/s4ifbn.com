#include #include #include using namespace std; // Structure to represent a process struct Process { int id; // process ID int arrivalTime; // time at which the process arrives in the ready queue int burstTime; // time required by the process to complete its execution }; // Function to implement FCFS scheduling algorithm void schedule(vector& processes) { // Sort the processes in ascending order of their arrival times sort(processes.begin(), processes.end(), [](const Process& p1, const Process& p2) { return p1.arrivalTime < p2.arrivalTime; }); // Initialize the current time to 0 int currentTime = 0; // Loop through each process in the sorted list for (const auto& process : processes) { // If the process has not arrived yet, skip to its arrival time if (currentTime < process.arrivalTime) { currentTime = process.arrivalTime; } // Execute the process for its full burst time currentTime += process.burstTime; // Print the process ID and the time at which it was executed cout << "Process " << process.id << " executed at time " << currentTime << endl; } } int main() { // Create a vector of processes vector processes = { {1, 0, 3}, // process 1 arrives at time 0 and requires 3 time units to execute {2, 1, 2}, // process 2 arrives at time 1 and requires 2 time units to execute {3, 2, 4}, // process 3 arrives at time 2 and requires 4 time units to execute {4, 3, 5}, // process 4 arrives at time 3 and requires 5 time units to execute {5, 4, 6} // process 5 arrives at time 4 and requires 6 time units to execute }; // Call the schedule function to execute the processes in FCFS order schedule(processes); return 0; }
